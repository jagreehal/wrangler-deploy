**Goal:** Make wrangler-deploy the goto library for Cloudflare Workers — graph visibility, orchestrated local dev, first-class CI/CD, and gradual config management.

**Architecture:** Extend the existing core engine (graph.ts, apply.ts, deploy.ts) with new modules: rich graph model with renderers, dev orchestrator using child processes, CI provider abstraction with GitHub implementation, and managed-mode config writer. All new features build on top of the existing dependency graph and state management.

**Tech Stack:** TypeScript, vitest + executable-stories-vitest, wrangler CLI (child process), GitHub API (for CI features), JSONC round-trip parsing.

import type { CfStageConfig, StageState } from "../types.js";

export type RichNodeType = "worker" | "kv" | "queue" | "d1" | "hyperdrive" | "r2" | "vectorize";

export interface RichNode {
  id: string;
  type: RichNodeType;
  label: string;
  resourceId?: string;
  status?: "active" | "missing" | "drifted" | "orphaned";
  deployedName?: string;
}

export type RichEdgeType = "service-binding" | "binding" | "producer" | "consumer" | "dead-letter";

export interface RichEdge {
  from: string;
  to: string;
  type: RichEdgeType;
  label?: string;
}

export interface RichGraph {
  nodes: RichNode[];
  edges: RichEdge[];
}

export function buildRichGraph(config: CfStageConfig, state?: StageState): RichGraph {
  const nodes: RichNode[] = [];
  const edges: RichEdge[] = [];

  for (const workerPath of config.workers) {
    const node: RichNode = {
      id: workerPath,
      type: "worker",
      label: workerPath.split("/").pop() ?? workerPath,
    };
    if (state?.workers[workerPath]) {
      node.deployedName = state.workers[workerPath].name;
    }
    nodes.push(node);
  }

  for (const [logicalName, resource] of Object.entries(config.resources)) {
    const node: RichNode = {
      id: logicalName,
      type: resource.type,
      label: logicalName,
    };
    if (state?.resources[logicalName]) {
      const rs = state.resources[logicalName];
      node.resourceId = rs.observed.id;
      node.status = rs.observed.status;
    }
    nodes.push(node);

    for (const [workerPath, binding] of Object.entries(resource.bindings)) {
      if (resource.type === "queue") {
        if (typeof binding === "object" && binding !== null) {
          const qb = binding as Record<string, unknown>;
          if (qb.producer) {
            edges.push({ from: workerPath, to: logicalName, type: "producer", label: qb.producer as string });
          }
          if (qb.consumer) {
            edges.push({ from: workerPath, to: logicalName, type: "consumer" });
          }
          if (qb.deadLetterFor) {
            edges.push({ from: qb.deadLetterFor as string, to: logicalName, type: "dead-letter" });
          }
        }
      } else {
        const bindingName = typeof binding === "string" ? binding : undefined;
        edges.push({ from: workerPath, to: logicalName, type: "binding", label: bindingName });
      }
    }
  }

  if (config.serviceBindings) {
    for (const [workerPath, bindings] of Object.entries(config.serviceBindings)) {
      for (const [bindingName, targetWorker] of Object.entries(bindings)) {
        edges.push({ from: workerPath, to: targetWorker, type: "service-binding", label: bindingName });
      }
    }
  }

  return { nodes, edges };
}
```

import type { RichGraph, RichNode, RichEdge } from "../graph-model.js";

const TYPE_ICONS: Record<string, string> = {
  worker: "⚙",
  kv: "📦",
  d1: "🗄",
  queue: "📨",
  hyperdrive: "🔌",
  r2: "🪣",
  vectorize: "🔍",
};

function formatNode(node: RichNode): string {
  const icon = TYPE_ICONS[node.type] ?? "•";
  let line = `${icon} ${node.label}`;
  if (node.deployedName) line += ` (${node.deployedName})`;
  if (node.resourceId) line += ` [${node.resourceId}]`;
  if (node.status && node.status !== "active") line += ` ⚠ ${node.status}`;
  return line;
}

function getEdgesFrom(edges: RichEdge[], nodeId: string): RichEdge[] {
  return edges.filter((e) => e.from === nodeId);
}

export function renderAscii(graph: RichGraph): string {
  const lines: string[] = [];
  const workers = graph.nodes.filter((n) => n.type === "worker");
  const resourceMap = new Map(graph.nodes.filter((n) => n.type !== "worker").map((n) => [n.id, n]));

  for (let i = 0; i < workers.length; i++) {
    const worker = workers[i];
    const isLast = i === workers.length - 1;
    const prefix = isLast ? "└── " : "├── ";
    const childPrefix = isLast ? "    " : "│   ";

    lines.push(`${prefix}${formatNode(worker)}`);

    const outEdges = getEdgesFrom(graph.edges, worker.id);
    for (let j = 0; j < outEdges.length; j++) {
      const edge = outEdges[j];
      const edgeIsLast = j === outEdges.length - 1;
      const edgePrefix = edgeIsLast ? "└── " : "├── ";
      const targetNode = graph.nodes.find((n) => n.id === edge.to);
      const targetLabel = targetNode ? formatNode(targetNode) : edge.to;
      const edgeLabel = edge.label ? `${edge.type}:${edge.label}` : edge.type;
      lines.push(`${childPrefix}${edgePrefix}${edgeLabel} → ${targetLabel}`);
    }
  }

  // Show resources with no worker edges (orphaned or standalone)
  for (const [id, node] of resourceMap) {
    const hasIncoming = graph.edges.some((e) => e.to === id);
    if (!hasIncoming) {
      lines.push(`└── ${formatNode(node)} (unbound)`);
    }
  }

  return lines.join("\n");
}

import type { RichGraph, RichNode, RichEdge, RichNodeType } from "../graph-model.js";

function sanitizeId(id: string): string {
  return id.replace(/[^a-zA-Z0-9]/g, "_");
}

function nodeShape(node: RichNode): string {
  const sid = sanitizeId(node.id);
  const label = node.deployedName ?? node.label;
  switch (node.type) {
    case "worker":
      return `${sid}([${label}])`;
    case "kv":
    case "r2":
      return `${sid}[(${label})]`;
    case "d1":
      return `${sid}[(${label})]`;
    case "queue":
      return `${sid}[/${label}\\]`;
    case "hyperdrive":
      return `${sid}{{${label}}}`;
    case "vectorize":
      return `${sid}>${label}]`;
    default:
      return `${sid}[${label}]`;
  }
}

const EDGE_STYLES: Record<string, string> = {
  "service-binding": "-->",
  binding: "-.->",
  producer: "-->",
  consumer: "-->",
  "dead-letter": "-. DLQ .->",
};

function edgeLine(edge: RichEdge): string {
  const from = sanitizeId(edge.from);
  const to = sanitizeId(edge.to);
  const arrow = EDGE_STYLES[edge.type] ?? "-->";
  if (edge.label) {
    return `  ${from} ${arrow}|${edge.label}| ${to}`;
  }
  return `  ${from} ${arrow} ${to}`;
}

function subgraphForType(type: RichNodeType): string | null {
  const names: Record<string, string> = {
    worker: "Workers",
    kv: "KV Namespaces",
    d1: "D1 Databases",
    queue: "Queues",
    hyperdrive: "Hyperdrive",
    r2: "R2 Buckets",
    vectorize: "Vectorize",
  };
  return names[type] ?? null;
}

export function renderMermaid(graph: RichGraph): string {
  const lines: string[] = ["graph TD"];

  // Group nodes by type into subgraphs
  const byType = new Map<RichNodeType, RichNode[]>();
  for (const node of graph.nodes) {
    const group = byType.get(node.type) ?? [];
    group.push(node);
    byType.set(node.type, group);
  }

  for (const [type, nodes] of byType) {
    const name = subgraphForType(type);
    if (name && nodes.length > 0) {
      lines.push(`  subgraph ${name}`);
      for (const node of nodes) {
        lines.push(`    ${nodeShape(node)}`);
      }
      lines.push("  end");
    }
  }

  // Edges
  for (const edge of graph.edges) {
    lines.push(edgeLine(edge));
  }

  return lines.join("\n");
}
```

import type { RichGraph, RichNode } from "../graph-model.js";

const SHAPES: Record<string, string> = {
  worker: "box",
  kv: "cylinder",
  d1: "cylinder",
  queue: "parallelogram",
  hyperdrive: "hexagon",
  r2: "folder",
  vectorize: "diamond",
};

function dotNode(node: RichNode): string {
  const shape = SHAPES[node.type] ?? "box";
  const label = node.deployedName ?? node.label;
  const extra = node.resourceId ? `\\n${node.resourceId}` : "";
  return `  "${node.id}" [label="${label}${extra}", shape=${shape}];`;
}

export function renderDot(graph: RichGraph): string {
  const lines: string[] = ["digraph {", "  rankdir=TB;"];

  for (const node of graph.nodes) {
    lines.push(dotNode(node));
  }

  for (const edge of graph.edges) {
    const label = edge.label ? ` [label="${edge.label}"]` : "";
    const style = edge.type === "dead-letter" ? ' [style=dashed, label="DLQ"]' : label;
    lines.push(`  "${edge.from}" -> "${edge.to}"${style};`);
  }

  lines.push("}");
  return lines.join("\n");
}

import type { RichGraph, RichEdge } from "./graph-model.js";

export interface UpstreamDep {
  id: string;
  type: string;
  sharedWith: string[];
}

export interface DownstreamDep {
  id: string;
  relationship: string;
  label?: string;
}

export interface ImpactResult {
  target: string;
  upstream: UpstreamDep[];
  downstream: DownstreamDep[];
  consequences: string[];
}

export function analyzeImpact(graph: RichGraph, targetId: string): ImpactResult {
  const node = graph.nodes.find((n) => n.id === targetId);
  if (!node) throw new Error(`"${targetId}" not found in graph`);

  // Upstream: resources this node depends on (outgoing edges from target)
  const outgoing = graph.edges.filter((e) => e.from === targetId);
  const upstream: UpstreamDep[] = [];

  for (const edge of outgoing) {
    const targetNode = graph.nodes.find((n) => n.id === edge.to);
    if (!targetNode || targetNode.type === "worker") continue;

    // Find other workers that also bind to this resource
    const otherEdges = graph.edges.filter(
      (e) => e.to === edge.to && e.from !== targetId && graph.nodes.find((n) => n.id === e.from)?.type === "worker"
    );
    const sharedWith = [...new Set(otherEdges.map((e) => e.from))];

    // Avoid duplicates
    if (!upstream.some((u) => u.id === edge.to)) {
      upstream.push({ id: edge.to, type: targetNode.type, sharedWith });
    }
  }

  // Downstream: workers that depend on this node (incoming edges to target)
  const incoming = graph.edges.filter((e) => e.to === targetId);
  const downstream: DownstreamDep[] = [];

  for (const edge of incoming) {
    const sourceNode = graph.nodes.find((n) => n.id === edge.from);
    if (!sourceNode || sourceNode.type !== "worker") continue;

    downstream.push({
      id: edge.from,
      relationship: edge.type,
      label: edge.label,
    });
  }

  // Generate consequences
  const consequences: string[] = [];
  for (const dep of downstream) {
    const desc = dep.label
      ? `${dep.id} loses ${dep.label} ${dep.relationship} (calls will fail)`
      : `${dep.id} loses ${dep.relationship} to ${targetId}`;
    consequences.push(desc);
  }

  // Check workers not in downstream — they're unaffected
  const workerNodes = graph.nodes.filter((n) => n.type === "worker" && n.id !== targetId);
  for (const w of workerNodes) {
    if (!downstream.some((d) => d.id === w.id)) {
      consequences.push(`${w.id} is unaffected (no direct dependency)`);
    }
  }

  return { target: targetId, upstream, downstream, consequences };
}

import type { StageState } from "../types.js";

export interface ResourceDiff {
  name: string;
  type: string;
  status: "only-in-a" | "only-in-b" | "same" | "different";
  idA?: string;
  idB?: string;
}

export interface WorkerDiff {
  path: string;
  status: "only-in-a" | "only-in-b" | "same";
  nameA?: string;
  nameB?: string;
}

export interface SecretDiff {
  worker: string;
  name: string;
  inA: "set" | "missing" | "absent";
  inB: "set" | "missing" | "absent";
}

export interface StageDiffResult {
  stageA: string;
  stageB: string;
  resources: ResourceDiff[];
  workers: WorkerDiff[];
  secrets: SecretDiff[];
}

export function diffStages(a: StageState, b: StageState): StageDiffResult {
  const resources: ResourceDiff[] = [];
  const allResourceNames = new Set([...Object.keys(a.resources), ...Object.keys(b.resources)]);

  for (const name of allResourceNames) {
    const inA = a.resources[name];
    const inB = b.resources[name];

    if (inA && !inB) {
      resources.push({ name, type: inA.type, status: "only-in-a", idA: inA.observed.id });
    } else if (!inA && inB) {
      resources.push({ name, type: inB.type, status: "only-in-b", idB: inB.observed.id });
    } else if (inA && inB) {
      const same = inA.observed.id === inB.observed.id && inA.observed.status === inB.observed.status;
      resources.push({
        name,
        type: inA.type,
        status: same ? "same" : "different",
        idA: inA.observed.id,
        idB: inB.observed.id,
      });
    }
  }

  const workers: WorkerDiff[] = [];
  const allWorkerPaths = new Set([...Object.keys(a.workers), ...Object.keys(b.workers)]);

  for (const path of allWorkerPaths) {
    const inA = a.workers[path];
    const inB = b.workers[path];

    if (inA && !inB) {
      workers.push({ path, status: "only-in-a", nameA: inA.name });
    } else if (!inA && inB) {
      workers.push({ path, status: "only-in-b", nameB: inB.name });
    } else if (inA && inB) {
      workers.push({ path, status: "same", nameA: inA.name, nameB: inB.name });
    }
  }

  const secrets: SecretDiff[] = [];
  const allWorkers = new Set([...Object.keys(a.secrets ?? {}), ...Object.keys(b.secrets ?? {})]);

  for (const worker of allWorkers) {
    const secretsA = a.secrets?.[worker] ?? {};
    const secretsB = b.secrets?.[worker] ?? {};
    const allNames = new Set([...Object.keys(secretsA), ...Object.keys(secretsB)]);

    for (const name of allNames) {
      const inA = secretsA[name] ?? "absent";
      const inB = secretsB[name] ?? "absent";
      if (inA !== inB) {
        secrets.push({ worker, name, inA: inA as "set" | "missing" | "absent", inB: inB as "set" | "missing" | "absent" });
      }
    }
  }

  return { stageA: a.stage, stageB: b.stage, resources, workers, secrets };
}

import type { CfStageConfig } from "../types.js";
import { resolveDeployOrder } from "./graph.js";

export function assignPorts(
  config: CfStageConfig,
  basePort: number,
  overrides?: Record<string, number>
): Record<string, number> {
  const order = resolveDeployOrder(config);
  const ports: Record<string, number> = {};
  const usedPorts = new Set(Object.values(overrides ?? {}));

  // Apply overrides first
  for (const [path, port] of Object.entries(overrides ?? {})) {
    ports[path] = port;
  }

  // Auto-assign remaining in dependency order
  let nextPort = basePort;
  for (const workerPath of order) {
    if (ports[workerPath] !== undefined) continue;

    while (usedPorts.has(nextPort)) {
      nextPort++;
    }
    ports[workerPath] = nextPort;
    usedPorts.add(nextPort);
    nextPort++;
  }

  return ports;
}

  "\x1b[36m", // cyan

export interface LogMultiplexer {

export function createLogMultiplexer(output: (line: string) => void): LogMultiplexer {
  let colorIndex = 0;
  const workerColors = new Map<string, string>();

  return {
    createWriter(workerPath: string) {
      const label = workerPath.split("/").pop() ?? workerPath;
      const color = COLORS[colorIndex % COLORS.length];
      workerColors.set(workerPath, color);
      colorIndex++;

      const maxLabelLen = 20;
      const paddedLabel = label.padEnd(maxLabelLen);
      const prefix = `${color}[${paddedLabel}]${RESET} `;

      return (data: string) => {
        const lines = data.split("\n").filter((l) => l.length > 0);
        for (const line of lines) {
          output(`${prefix}${line}`);
        }
      };
    },
  };
}

import { spawn, type ChildProcess } from "node:child_process";
import { resolve } from "node:path";
import type { CfStageConfig } from "../types.js";
import { resolveDeployOrder } from "./graph.js";
import { assignPorts } from "./dev-ports.js";
import { createLogMultiplexer } from "./dev-logs.js";

export interface WorkerDevPlan {
  workerPath: string;
  cwd: string;
  port: number;
  args: string[];
}

export interface DevPlan {
  workers: WorkerDevPlan[];
  ports: Record<string, number>;
}

export interface DevOptions {
  basePort?: number;
  filter?: string;
  workerOptions?: Record<string, { devPort?: number; devArgs?: string[] }>;
}

export function buildDevPlan(config: CfStageConfig, rootDir: string, options: DevOptions): DevPlan {
  const basePort = options.basePort ?? 8787;
  let deployOrder = resolveDeployOrder(config);

  // Filter to target + transitive dependencies
  if (options.filter) {
    const needed = new Set<string>();
    const addWithDeps = (path: string) => {
      if (needed.has(path)) return;
      needed.add(path);
      // Add service binding dependencies
      const bindings = config.serviceBindings?.[path] ?? {};
      for (const target of Object.values(bindings)) {
        addWithDeps(target);
      }
    };
    addWithDeps(options.filter);
    deployOrder = deployOrder.filter((p) => needed.has(p));
  }

  // Build port overrides from workerOptions
  const portOverrides: Record<string, number> = {};
  for (const [path, opts] of Object.entries(options.workerOptions ?? {})) {
    if (opts.devPort) portOverrides[path] = opts.devPort;
  }

  const ports = assignPorts(
    { ...config, workers: deployOrder },
    basePort,
    portOverrides
  );

  const workers: WorkerDevPlan[] = deployOrder.map((workerPath) => {
    const port = ports[workerPath];
    const customArgs = options.workerOptions?.[workerPath]?.devArgs ?? [];
    const args = ["dev", "--port", String(port), ...customArgs];

    return {
      workerPath,
      cwd: resolve(rootDir, workerPath),
      port,
      args,
    };
  });

  return { workers, ports };
}

export interface DevHandle {
  stop(): void;
  processes: Map<string, ChildProcess>;
}

export function startDev(plan: DevPlan, options?: { output?: (line: string) => void }): DevHandle {
  const output = options?.output ?? ((line: string) => console.log(line));
  const mux = createLogMultiplexer(output);
  const processes = new Map<string, ChildProcess>();

  // Print port map
  output("");
  output("  wrangler-deploy dev\n");
  for (const w of plan.workers) {
    output(`  ${w.workerPath.padEnd(30)} → localhost:${w.port}`);
  }
  output("");

  // Start workers in order
  for (const w of plan.workers) {
    const write = mux.createWriter(w.workerPath);
    const child = spawn("npx", ["wrangler", ...w.args], {
      cwd: w.cwd,
      stdio: ["ignore", "pipe", "pipe"],
      env: { ...process.env },
    });

    child.stdout?.on("data", (data: Buffer) => write(data.toString()));
    child.stderr?.on("data", (data: Buffer) => write(data.toString()));

    child.on("exit", (code) => {
      write(`process exited with code ${code}`);
      processes.delete(w.workerPath);
    });

    processes.set(w.workerPath, child);
  }

  return {
    processes,
    stop() {
      for (const [, child] of processes) {
        child.kill("SIGTERM");
      }
    },
  };
}

/**
 * Update a JSONC string with new values, preserving comments and formatting.
 *
 * Strategy: parse the JSONC to get the current object, deep-merge the updates,
 * then serialize back. Comments are preserved by tracking their positions
 * relative to keys and reinserting them.
 */

interface CommentEntry {
  line: number;
  text: string;
}

function stripComments(jsonc: string): { json: string; comments: CommentEntry[] } {
  const comments: CommentEntry[] = [];
  const lines = jsonc.split("\n");
  const cleanLines: string[] = [];

  for (let i = 0; i < lines.length; i++) {
    const line = lines[i];
    // Full-line comment
    const trimmed = line.trim();
    if (trimmed.startsWith("//")) {
      comments.push({ line: i, text: line });
      cleanLines.push("");
      continue;
    }
    // Inline comment — strip but keep the JSON part
    const inString = false;
    let cleaned = "";
    let j = 0;
    let inStr = false;
    let escaped = false;
    while (j < line.length) {
      const ch = line[j];
      if (escaped) { cleaned += ch; escaped = false; j++; continue; }
      if (ch === "\\") { cleaned += ch; escaped = true; j++; continue; }
      if (ch === '"') { inStr = !inStr; cleaned += ch; j++; continue; }
      if (!inStr && ch === "/" && line[j + 1] === "/") {
        comments.push({ line: i, text: line.slice(j) });
        break;
      }
      if (!inStr && ch === "/" && line[j + 1] === "*") {
        // Block comment on single line
        const end = line.indexOf("*/", j + 2);
        if (end !== -1) {
          comments.push({ line: i, text: line.slice(j, end + 2) });
          j = end + 2;
          continue;
        }
      }
      cleaned += ch;
      j++;
    }
    cleanLines.push(cleaned);
  }

  // Strip trailing commas before } or ]
  let json = cleanLines.join("\n");
  json = json.replace(/,(\s*[}\]])/g, "$1");

  return { json, comments };
}

function deepMerge(target: Record<string, unknown>, source: Record<string, unknown>): Record<string, unknown> {
  const result = { ...target };
  for (const [key, value] of Object.entries(source)) {
    if (value !== null && typeof value === "object" && !Array.isArray(value) &&
        result[key] !== null && typeof result[key] === "object" && !Array.isArray(result[key])) {
      result[key] = deepMerge(result[key] as Record<string, unknown>, value as Record<string, unknown>);
    } else {
      result[key] = value;
    }
  }
  return result;
}

export function updateJsonc(jsonc: string, updates: Record<string, unknown>): string {
  const { json, comments } = stripComments(jsonc);
  const parsed = JSON.parse(json);
  const merged = deepMerge(parsed, updates);
  const serialized = JSON.stringify(merged, null, 2);

  // Reinsert comments at their original line positions
  const outputLines = serialized.split("\n");
  for (const comment of comments) {
    if (comment.line < outputLines.length) {
      // Insert comment before the line at its original position
      outputLines.splice(comment.line, 0, comment.text);
    } else {
      outputLines.push(comment.text);
    }
  }

  return outputLines.join("\n");
}

import type { CfStageConfig, StageState } from "../types.js";

export interface ManagedEnvSection {
  kv_namespaces?: Array<{ binding: string; id: string }>;
  d1_databases?: Array<{ binding: string; database_id: string; database_name: string }>;
  hyperdrive?: Array<{ binding: string; id: string }>;
  queues?: { producers?: Array<{ binding: string; queue: string }> };
  r2_buckets?: Array<{ binding: string; bucket_name: string }>;
  services?: Array<{ binding: string; service: string }>;
}

export function writeManagedBindings(
  workerPath: string,
  config: CfStageConfig,
  state: StageState
): ManagedEnvSection {
  const section: ManagedEnvSection = {};

  for (const [logicalName, resource] of Object.entries(config.resources)) {
    const binding = resource.bindings[workerPath];
    if (!binding) continue;

    const resourceState = state.resources[logicalName];
    if (!resourceState?.observed.id) continue;

    const id = resourceState.observed.id;
    const desiredName = resourceState.desired.name;

    switch (resource.type) {
      case "kv": {
        const bindingName = typeof binding === "string" ? binding : undefined;
        if (!bindingName) break;
        section.kv_namespaces ??= [];
        section.kv_namespaces.push({ binding: bindingName, id });
        break;
      }
      case "d1": {
        const bindingName = typeof binding === "string" ? binding : undefined;
        if (!bindingName) break;
        section.d1_databases ??= [];
        section.d1_databases.push({ binding: bindingName, database_id: id, database_name: desiredName });
        break;
      }
      case "hyperdrive": {
        const bindingName = typeof binding === "string" ? binding : undefined;
        if (!bindingName) break;
        section.hyperdrive ??= [];
        section.hyperdrive.push({ binding: bindingName, id });
        break;
      }
      case "queue": {
        if (typeof binding === "object" && binding !== null) {
          const qb = binding as Record<string, unknown>;
          if (qb.producer && typeof qb.producer === "string") {
            section.queues ??= {};
            section.queues.producers ??= [];
            section.queues.producers.push({ binding: qb.producer, queue: desiredName });
          }
        }
        break;
      }
      case "r2": {
        const bindingName = typeof binding === "string" ? binding : undefined;
        if (!bindingName) break;
        section.r2_buckets ??= [];
        section.r2_buckets.push({ binding: bindingName, bucket_name: desiredName });
        break;
      }
    }
  }

  // Service bindings
  if (config.serviceBindings?.[workerPath]) {
    section.services = [];
    for (const [bindingName, targetWorker] of Object.entries(config.serviceBindings[workerPath])) {
      const targetState = state.workers[targetWorker];
      if (targetState) {
        section.services.push({ binding: bindingName, service: targetState.name });
      }
    }
  }

  return section;
}

import type { CfStageConfig } from "../types.js";

export function validateConfig(config: CfStageConfig): string[] {
  const errors: string[] = [];
  const workerSet = new Set(config.workers);
  const resourceSet = new Set(Object.keys(config.resources));

  // Check resource bindings reference valid workers
  for (const [resourceName, resource] of Object.entries(config.resources)) {
    for (const workerPath of Object.keys(resource.bindings)) {
      if (!workerSet.has(workerPath)) {
        errors.push(`Resource "${resourceName}" has binding to unknown worker "${workerPath}"`);
      }
    }

    // Check DLQ references
    if (resource.type === "queue") {
      for (const binding of Object.values(resource.bindings)) {
        if (typeof binding === "object" && binding !== null) {
          const qb = binding as Record<string, unknown>;
          if (qb.deadLetterFor && typeof qb.deadLetterFor === "string") {
            if (!resourceSet.has(qb.deadLetterFor)) {
              errors.push(`DLQ "${resourceName}" references non-existent queue "${qb.deadLetterFor}"`);
            }
          }
        }
      }
    }
  }

  // Check service bindings reference valid workers
  if (config.serviceBindings) {
    for (const [source, bindings] of Object.entries(config.serviceBindings)) {
      if (!workerSet.has(source)) {
        errors.push(`Service binding source "${source}" is not in workers list`);
      }
      for (const [bindingName, target] of Object.entries(bindings)) {
        if (!workerSet.has(target)) {
          errors.push(`Service binding "${bindingName}" in "${source}" targets unknown worker "${target}"`);
        }
      }
    }

    // Check for circular service bindings
    const visited = new Set<string>();
    const inStack = new Set<string>();

    function hasCycle(node: string): boolean {
      if (inStack.has(node)) return true;
      if (visited.has(node)) return false;
      visited.add(node);
      inStack.add(node);

      const bindings = config.serviceBindings?.[node] ?? {};
      for (const target of Object.values(bindings)) {
        if (hasCycle(target)) {
          errors.push(`Circular service binding detected involving "${node}" and "${target}"`);
          inStack.delete(node);
          return true;
        }
      }

      inStack.delete(node);
      return false;
    }

    for (const worker of config.workers) {
      hasCycle(worker);
    }
  }

  return errors;
}

import type { CfStageConfig } from "../types.js";

export interface DoctorCheck {
  name: string;
  status: "pass" | "warn" | "fail";
  message: string;
  details?: string;
}

export interface DoctorDeps {
  wranglerVersion: () => string;
  wranglerAuth: () => string;
  workerExists: (path: string) => boolean;
  configErrors: string[];
}

export function runDoctor(config: CfStageConfig, deps: DoctorDeps): DoctorCheck[] {
  const checks: DoctorCheck[] = [];

  // 1. Wrangler installed
  try {
    const version = deps.wranglerVersion();
    checks.push({ name: "wrangler", status: "pass", message: `wrangler ${version} installed` });
  } catch {
    checks.push({
      name: "wrangler",
      status: "fail",
      message: "wrangler not found",
      details: "Install with: npm install -g wrangler",
    });
  }

  // 2. Wrangler authenticated
  try {
    const user = deps.wranglerAuth();
    checks.push({ name: "auth", status: "pass", message: `Authenticated as ${user}` });
  } catch {
    checks.push({
      name: "auth",
      status: "fail",
      message: "Not authenticated",
      details: "Run: wrangler login",
    });
  }

  // 3. Worker paths exist
  const missingPaths = config.workers.filter((p) => !deps.workerExists(p));
  if (missingPaths.length === 0) {
    checks.push({ name: "worker-paths", status: "pass", message: "All worker paths exist" });
  } else {
    checks.push({
      name: "worker-paths",
      status: "fail",
      message: `${missingPaths.length} worker path(s) not found`,
      details: missingPaths.join(", "),
    });
  }

  // 4. Config validation
  if (deps.configErrors.length === 0) {
    checks.push({ name: "config", status: "pass", message: "Config is valid" });
  } else {
    checks.push({
      name: "config",
      status: "fail",
      message: `${deps.configErrors.length} config error(s)`,
      details: deps.configErrors.join("\n"),
    });
  }

  return checks;
}

  "init", "introspect", "plan", "apply", "deploy", "destroy",
  "gc", "status", "secrets", "verify", "graph", "impact", "diff",
  "dev", "ci", "doctor", "completions",

const FLAGS = ["--stage", "--format", "--force", "--verify", "--filter", "--port", "--provider", "--branch", "--verbose"];

export function generateCompletions(shell: "zsh" | "bash" | "fish"): string {
  switch (shell) {
    case "zsh":
      return `#compdef wd wrangler-deploy

_wd() {
  local commands=(${COMMANDS.map((c) => `'${c}:${c} command'`).join(" ")})
  local flags=(${FLAGS.map((f) => `'${f}'`).join(" ")})

  _arguments -C \\
    '1:command:->cmd' \\
    '*::arg:->args'

  case "$state" in
    cmd)
      _describe 'command' commands
      ;;
    args)
      _values 'flags' $flags
      ;;
  esac
}

compdef _wd wd
compdef _wd wrangler-deploy
`;

    case "bash":
      return `_wd_completions() {
  local commands="${COMMANDS.join(" ")}"
  local flags="${FLAGS.join(" ")}"

  if [ "$COMP_CWORD" -eq 1 ]; then
    COMPREPLY=($(compgen -W "$commands" -- "\${COMP_WORDS[COMP_CWORD]}"))
  else
    COMPREPLY=($(compgen -W "$flags" -- "\${COMP_WORDS[COMP_CWORD]}"))
  fi
}

complete -F _wd_completions wd
complete -F _wd_completions wrangler-deploy
`;

    case "fish":
      return COMMANDS.map(
        (c) => `complete -c wd -n '__fish_use_subcommand' -a '${c}' -d '${c} command'`
      ).join("\n") + "\n" +
      FLAGS.map(
        (f) => `complete -c wd -l '${f.replace("--", "")}' -d '${f}'`
      ).join("\n") + "\n";
  }

export function createTimer(): { elapsed(): string } {
  const start = performance.now();
  return {
    elapsed(): string {
      const ms = performance.now() - start;
      if (ms < 1000) return `${Math.round(ms)}ms`;
      if (ms < 60_000) return `${(ms / 1000).toFixed(1)}s`;
      const mins = Math.floor(ms / 60_000);
      const secs = ((ms % 60_000) / 1000).toFixed(0);
      return `${mins}m${secs}s`;
    },
  };
}
